# E2E Testing Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Playwright E2E 테스트로 전체 사용자 여정(홈→측정→결과→CES→플레이어)을 커버한다.

**Architecture:** 페이지별 독립 테스트(시드 데이터 기반) + 통합 골든패스 테스트(실제 흐름). 공통 헬퍼로 HashRouter 네비게이션과 localStorage 주입을 추상화.

**Tech Stack:** Playwright, TypeScript, Chromium + WebKit

- [ ] **Step 3: setup.ts 헬퍼 작성**

```ts
// e2e/helpers/setup.ts
import { Page } from '@playwright/test';
import { TEST_PATIENT, TEST_SESSION, TEST_HISTORY, SEED_KEYS } from '../fixtures/test-data';

/**
 * HashRouter 경로로 이동
 */
export async function navigateTo(page: Page, path: string) {
  await page.goto(`/#${path}`);
  await page.waitForLoadState('networkidle');
}

/**
 * 기본 시드 데이터 주입 (환자 + 세션)
 */
export async function seedDefault(page: Page) {
  await page.goto('/#/');
  await page.evaluate(({ patient, session, keys }) => {
    localStorage.setItem(keys.session, JSON.stringify(session));
    localStorage.setItem(keys.patients, JSON.stringify([patient]));
  }, { patient: TEST_PATIENT, session: TEST_SESSION, keys: SEED_KEYS });
}

/**
 * 히스토리 포함 시드 데이터 주입 (경과관찰용)
 */
export async function seedWithHistory(page: Page) {
  await page.goto('/#/');
  await page.evaluate(({ patient, session, history, keys }) => {
    localStorage.setItem(keys.session, JSON.stringify(session));
    localStorage.setItem(keys.patients, JSON.stringify([patient]));
    localStorage.setItem(keys.history(patient.id), JSON.stringify(history));
  }, { patient: TEST_PATIENT, session: TEST_SESSION, history: TEST_HISTORY, keys: SEED_KEYS });
}

/**
 * localStorage 전체 초기화
 */
export async function clearStorage(page: Page) {
  await page.evaluate(() => localStorage.clear());
}
```

```ts
import { test, expect } from '@playwright/test';
import { navigateTo, clearStorage } from '../helpers/setup';

test.describe('홈 페이지', () => {
  test.beforeEach(async ({ page }) => {
    await clearStorage(page);
    await navigateTo(page, '/');
  });

  test('페이지 타이틀과 기본 UI 렌더링', async ({ page }) => {
    await expect(page.getByText('ROM 측정 시스템')).toBeVisible();

  });

  test('환자 이름/나이 입력', async ({ page }) => {
    await page.getByPlaceholder('성함').fill('홍길동');
    await page.getByPlaceholder('세').fill('30');
    await expect(page.getByPlaceholder('성함')).toHaveValue('홍길동');
    await expect(page.getByPlaceholder('세')).toHaveValue('30');
  });

  test('사이드 모드 선택', async ({ page }) => {
    await page.getByRole('button', { name: '좌측만' }).click();
    await page.getByRole('button', { name: '우측만' }).click();
    await page.getByRole('button', { name: '양쪽' }).click();
  });

  test('관절 선택 후 측정 시작 버튼 활성화', async ({ page }) => {
    // 환자 정보 입력

    await page.getByPlaceholder('세').fill('30');

    // 관절 토글 (어깨)
    await page.getByText('어깨').click();

    // "측정 시작하기" 버튼이 활성화되어야 함
    const startBtn = page.getByRole('button', { name: /측정 시작하기/ });

  });
});
```

```ts
import { test, expect } from '@playwright/test';
import { seedDefault, navigateTo } from '../helpers/setup';

test.describe('CES 플레이어', () => {
  test.beforeEach(async ({ page }) => {
    await seedDefault(page);
    await navigateTo(page, '/ces');
    // CES 페이지에서 자동 재생 시작으로 플레이어 진입
    await page.getByRole('button', { name: /자동 재생/ }).click();
    await page.waitForURL(/.*#\/ces-player/);
  });

  test('스토리 프로그레스바 표시', async ({ page }) => {
    await expect(page.locator('.story-bar, [class*="story"]')).toBeVisible();
  });

  test('컨트롤러 버튼 존재 (재생, 이전, 다음)', async ({ page }) => {
    // 컨트롤러 영역에 버튼이 존재하는지 확인
    await expect(page.locator('button').first()).toBeVisible();
  });

  test('운동 이름 또는 휴식 표시', async ({ page }) => {
    // 운동명 또는 "휴식" 텍스트가 보여야 함
    const hasExercise = await page.getByText(/억제|신장|활성|통합|휴식/).first().isVisible();
    expect(hasExercise).toBeTruthy();
```

```ts
import { test, expect } from '@playwright/test';
import { seedDefault, navigateTo } from '../helpers/setup';

test.describe('설정 페이지', () => {
  test.beforeEach(async ({ page }) => {
    await seedDefault(page);
    await navigateTo(page, '/settings');
  });

  test('페이지 타이틀 렌더링', async ({ page }) => {
    await expect(page.getByText('설정')).toBeVisible();
  });

  test('데이터 통계 카드 표시', async ({ page }) => {
    await expect(page.getByText('등록 환자')).toBeVisible();
    await expect(page.getByText('측정 기록')).toBeVisible();
  });

  test('데이터 내보내기 버튼 존재', async ({ page }) => {
    await expect(page.getByRole('button', { name: /데이터 내보내기/ })).toBeVisible();
  });

  test('데이터 삭제 버튼 클릭 시 confirm 다이얼로그', async ({ page }) => {
    page.on('dialog', async (dialog) => {
      expect(dialog.type()).toBe('confirm');
      await dialog.dismiss(); // 취소
    });
    await page.getByRole('button', { name: /모든 환자 데이터 삭제/ }).click();
  });

  test('버전 정보 표시', async ({ page }) => {
    await expect(page.getByText(/버전/)).toBeVisible();
  });
});
```

### Task 11: 통합 테스트 — 환자 등록 → 측정 → 결과

**Files:**
- Create: `e2e/journeys/patient-to-results.spec.ts`

- [ ] **Step 1: 테스트 작성**

```ts
// e2e/journeys/patient-to-results.spec.ts
import { test, expect } from '@playwright/test';
import { clearStorage, navigateTo } from '../helpers/setup';

test.describe('골든패스: 환자 등록 → 측정 → 결과', () => {
  test.beforeEach(async ({ page }) => {
    await clearStorage(page);
    await navigateTo(page, '/');
  });

  test('환자 등록부터 결과 확인까지 전체 흐름', async ({ page }) => {
    // 1. 홈에서 환자 정보 입력
    await page.getByPlaceholder('성함').fill('E2E테스트');
    await page.getByPlaceholder('세').fill('35');

    // 2. 관절 선택 (어깨)
    await page.getByText('어깨').click();

    // 3. 측정 시작
    await page.getByRole('button', { name: /측정 시작하기/ }).click();

    // 4. 측정 페이지 도착 확인
    await expect(page).toHaveURL(/.*#\/measure/);
    await expect(page.getByText(/어깨/)).toBeVisible();

    // 5. 측정값 입력 → 다음 반복 (모든 동작 완료까지)
    // 실제 DOM 구조에 맞게 입력 방식 조정 필요
    const nextBtn = page.getByRole('button', { name: /다음|계속|완료/ });
    while (await nextBtn.isVisible().catch(() => false)) {
      await nextBtn.click();
      await page.waitForTimeout(200);
    }

    // 6. 결과 페이지 도착 확인
    await expect(page).toHaveURL(/.*#\/results/);
    await expect(page.getByText('E2E테스트')).toBeVisible();
    await expect(page.getByText('측정 관절')).toBeVisible();
  });
});
```

```ts
// e2e/journeys/results-to-ces.spec.ts
import { test, expect } from '@playwright/test';
import { seedDefault, navigateTo } from '../helpers/setup';

test.describe('골든패스: 결과 → CES → 플레이어', () => {
  test.beforeEach(async ({ page }) => {
    await seedDefault(page);
    await navigateTo(page, '/results');
  });

  test('결과에서 CES 재활 → 플레이어까지 전체 흐름', async ({ page }) => {
    // 1. 결과 페이지 확인
    await expect(page.getByText('테스트환자')).toBeVisible();

    // 2. CES 재활 시작 클릭
    await page.getByRole('button', { name: /CES 재활 시작/ }).click();
    await expect(page).toHaveURL(/.*#\/ces/);

    // 3. CES 페이지에서 스테이지 탭 확인
    await expect(page.getByText('억제')).toBeVisible();
    await expect(page.getByText('신장')).toBeVisible();

    // 4. 자동 재생 시작
    await page.getByRole('button', { name: /자동 재생/ }).click();
    await expect(page).toHaveURL(/.*#\/ces-player/);

    // 5. 플레이어 동작 확인
    await page.waitForTimeout(500);
    // 프로그레스바 또는 운동 관련 UI가 보이는지 확인
    await expect(page.locator('.story-bar, [class*="story"], [class*="progress"]').first()).toBeVisible();
  });
});
```
